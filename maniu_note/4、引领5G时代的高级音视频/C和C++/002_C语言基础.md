# C语言基础

## struct

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    unsigned int a;// 无符号正数
    int b;
} Person;

int main() {
    Person *person = malloc(sizeof(Person));
    person->a = 44;
    person->b = 22;
    printf("a = %d\n", person->a);
    printf("b = %d\n", person->b);
    return 0;
}
```

## 指针

```c
#include <stdio.h>

int main() {
    int a = 10;
    int *p;// 定义一个地址的变量
    p = &a;// 把a的地址赋值给p

    printf("p = %d\n", *p);
    return 0;
}
```

```c
#include <stdio.h>

int main() {
    int c[9] = {1, 2, 3, 4, 5, 6, 7, 8, 9};

    // 数组是一个比较特殊的指针
    printf("%d \n",c);// 首地址
    printf("%d \n",&c);// 首地址的地址，还是首地址
    printf("%d \n",&c[0]);// 取首地址值的地址

    printf("%d \n",*c);// 取首地址的值
    printf("%d \n",c[0]);// 取首地址的值
    printf("%d \n",*(c+1));// 取第二个地址的值
    
    return 0;
}
```

```c
#include <stdio.h>

int main() {
    int a = 10;
    int b = 20;
    int c = 30;
    int *p1 = &a;
    int *p2 = &b;
    int *p3 = &c;
    int *arr[3] = {p1, p2, p3};// 数组指针：数组存放的是指针类型

    printf("%d \n", *arr[0]);// 10
    printf("%d \n", **arr);// 10

    return 0;
}
```

```c
#include <stdio.h>

int main() {
    int a[] = {10, 20};																			
    int b[] = {30, 10};
    int c[] = {10};
    int *arr[3] = {a, b, c};// 二维数组

    printf("%d \n", **arr);
    printf("%d \n", arr[1][1]);

    return 0;
}
```
